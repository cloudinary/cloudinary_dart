Asset raw(String publicId, [AssetBuilder? options]) {
  var builder = AssetBuilder()
    ..publicId(publicId)
    ..cloudConfig(config.cloudConfig)
    ..urlConfig(config.urlConfig)
    ..assetType('raw');
  if (options != null) {
    builder.combineWith(options);
  }
  return Asset.withBuilder(builder);
}
